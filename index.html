async function checkChzzkLive() { const grid = document.getElementById('live-grid'); grid.innerHTML = ''; let liveCount = 0; for (let m of db.members) { // 풀 URL이 입력되어 있어도 32자리 ID만 정확히 추출 let chzzkId = (m.chzzk_id || m.channel_id || m.chzzkId || '').trim(); if (chzzkId.includes('/')) { chzzkId = chzzkId.split('/').pop().split('?')[0]; } const name = (m.name || m.이름 || '').trim(); if (!chzzkId) continue; try { const targetApi = `https://api.chzzk.naver.com/polling/v2/channels/${chzzkId}/live-status`; // AllOrigins 안정적인 CORS 우회 프록시 사용 const proxyUrl = `https://api.allorigins.win/raw?url=${encodeURIComponent(targetApi)}`; let res = await fetch(proxyUrl); let data = await res.json(); // 만약 프록시 응답이 텍스트로 들어온 경우 JSON 파싱 if (typeof data === 'string') { data = JSON.parse(data); } if (data?.content?.status === 'OPEN') { liveCount++; const liveTitle = data.content.liveTitle || '라이브 방송 중'; grid.innerHTML += `
${name} LIVE

${liveTitle}

시청하기
`; } } catch (e) { console.error(`Live Check Error (${name}):`, e); } } if (liveCount === 0) { grid.innerHTML = `
현재 방송 중인 멤버가 없습니다.
`; } }
