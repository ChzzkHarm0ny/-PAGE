async function checkChzzkLive() { const grid = document.getElementById('live-grid'); grid.innerHTML = ''; let liveCount = 0; for (let m of db.members) { // chzzk_id, channel_id, 방송국링크 등 다양한 헤더 파싱 대응 const chzzkId = (m.chzzk_id || m.channel_id || (m.방송국링크 ? m.방송국링크.split('/').pop() : '') || '').trim(); const name = (m.name || m.이름 || '').trim(); if (!chzzkId) continue; try { // 💡 CORS 우회 프록시(corsproxy.io)를 경유하여 치지직 API 호출 const targetApi = `https://api.chzzk.naver.com/polling/v2/channels/${chzzkId}/live-status`; const proxyUrl = `https://corsproxy.io/?${encodeURIComponent(targetApi)}`; const res = await fetch(proxyUrl); const data = await res.json(); if (data?.content?.status === 'OPEN') { liveCount++; const liveTitle = data.content.liveTitle || '라이브 방송 중'; grid.innerHTML += `
${name} LIVE

${liveTitle}

시청하기
`; } } catch(e) { console.error(`Live Check Error (${name}):`, e); } } if (liveCount === 0) { grid.innerHTML = `
현재 방송 중인 멤버가 없습니다.
`; } }
